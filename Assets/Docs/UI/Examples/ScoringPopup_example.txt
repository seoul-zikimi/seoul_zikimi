// 프리팹 저장 경로: Assets/Resources/UI/Popup/ScoringPopup

using TMPro;
using UnityEngine;
using UnityEngine.UI;

public class ScoringPopup : UIPopup
{
    enum Texts   { Txt_TotalScore, Txt_Breakdown }
    enum Buttons { Btn_Confirm }

    public override void Init()
    {
        Bind<TextMeshProUGUI>(typeof(Texts));
        Bind<Button>(typeof(Buttons));

        BindEvent(Get<Button>((int)Buttons.Btn_Confirm).gameObject,
            _ => UIManager.Instance.ClosePopupUI());
    }

    public void SetScore(int total, string breakdown)
    {
        Get<TextMeshProUGUI>((int)Texts.Txt_TotalScore).text = $"{total}점";
        Get<TextMeshProUGUI>((int)Texts.Txt_Breakdown).text  = breakdown;
    }
}
